235 //--------------------------------------------------------------
236 // WS2812 Timing : (1.25us = 800 kHz)
237 // logische-0 => HI:0.35us , LO:0.90us
238 // logische-1 = HI:0.90us , LO:0.35us
239 //
240 // WS23812 Bit-Format : (8G8R8B)
241 // 24bit pro LED (30us pro LED)
242 // 8bit pro Farbe (MSB first)
243 // Farbreihenfolge : GRB
244 //
245 // nach jedem Frame von n-LEDs kommt eine Pause von >= 50us
246 //
247 // STM32F1xx
248 // Grundfrequenz (TIM3) = 2*APB1 (APB1=24MHz) => TIM_CLK=48MHz
249 // PWM-Frq = TIM_CLK/(periode+1)/(prescaler+1)
250 // periode : 0 bis 0xFFFF
251 // prescale : 0 bis 0xFFFF
252 //
253 //--------------------------------------------------------------
254 #define WS2812_TIM_PRESCALE 0 // F_T3 = 48 MHz (20.8ns)
255 #define WS2812_TIM_PERIODE 89 //89 // F_PWM = 800 kHz (1.25us)
256 #define WS2812_LO_TIME 28 // 29 * 11.9ns = 0.34us
257 #define WS2812_HI_TIME 70 // 76 * 11.9ns = 0.90us
258 //--------------------------------------------------------------
259 // defines vom WS2812 (nicht abaendern)
260 //--------------------------------------------------------------
261 #define WS2812_BIT_PER_LED 24 // 3*8bit pro LED
262 #define WS2812_PAUSE_ANZ 2 // fuer Pause (2*30us)
263 #define WS2812_TIMER_BUF_LEN (WS2812_LED_MAX_ANZ+WS2812_PAUSE_ANZ)*WS2812_BIT_PER_LED
