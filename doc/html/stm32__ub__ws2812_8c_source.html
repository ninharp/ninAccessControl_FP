361 void UB_WS2812_RGB_2_HSV(WS2812_HSV_t hsv_col, WS2812_RGB_t *rgb_col)
362 {
363  uint8_t diff;
364 
365  // Grenzwerte
366  if (hsv_col.h > 359)
367  hsv_col.h = 359;
368  if (hsv_col.s > 100)
369  hsv_col.s = 100;
370  if (hsv_col.v > 100)
371  hsv_col.v = 100;
372 
373  if (hsv_col.h < 61)
374  {
375  rgb_col->red = 255;
376  rgb_col->green = (425 * hsv_col.h) / 100;
377  rgb_col->blue = 0;
378  }
379  else if (hsv_col.h < 121)
380  {
381  rgb_col->red = 255 - ((425 * (hsv_col.h - 60)) / 100);
382  rgb_col->green = 255;
383  rgb_col->blue = 0;
384  }
385  else if (hsv_col.h < 181)
386  {
387  rgb_col->red = 0;
388  rgb_col->green = 255;
389  rgb_col->blue = (425 * (hsv_col.h - 120)) / 100;
390  }
391  else if (hsv_col.h < 241)
392  {
393  rgb_col->red = 0;
394  rgb_col->green = 255 - ((425 * (hsv_col.h - 180)) / 100);
395  rgb_col->blue = 255;
396  }
397  else if (hsv_col.h < 301)
398  {
399  rgb_col->red = (425 * (hsv_col.h - 240)) / 100;
400  rgb_col->green = 0;
401  rgb_col->blue = 255;
402  }
403  else
404  {
405  rgb_col->red = 255;
406  rgb_col->green = 0;
407  rgb_col->blue = 255 - ((425 * (hsv_col.h - 300)) / 100);
408  }
409 
410  hsv_col.s = 100 - hsv_col.s;
411  diff = ((255 - rgb_col->red) * hsv_col.s) / 100;
412  rgb_col->red = rgb_col->red + diff;
413  diff = ((255 - rgb_col->green) * hsv_col.s) / 100;
414  rgb_col->green = rgb_col->green + diff;
415  diff = ((255 - rgb_col->blue) * hsv_col.s) / 100;
416  rgb_col->blue = rgb_col->blue + diff;
417 
418  rgb_col->red = (rgb_col->red * hsv_col.v) / 100;
419  rgb_col->green = (rgb_col->green * hsv_col.v) / 100;
420  rgb_col->blue = (rgb_col->blue * hsv_col.v) / 100;
421 }
