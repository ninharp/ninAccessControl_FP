275 void I2C2_Start(uint8_t address, uint8_t direction)
276 {
277  I2C_TypeDef* I2Cx = I2C2;
278  // wait until I2C1 is not busy anymore
279  while(I2C_GetFlagStatus(I2Cx, I2C_FLAG_BUSY));
280 
281  // Send I2C1 START condition
282  I2C_GenerateSTART(I2Cx, ENABLE);
283 
284  // wait for I2C1 EV5 --> Slave has acknowledged start condition
285  while(!I2C_CheckEvent(I2Cx, I2C_EVENT_MASTER_MODE_SELECT));
286 
287  // Send slave Address for write
288  I2C_Send7bitAddress(I2Cx, address, direction);
289 
290  /* wait for I2C1 EV6, check if
291  * either Slave has acknowledged Master transmitter or
292  * Master receiver mode, depending on the transmission
293  * direction
294  */
295  if(direction == I2C_Direction_Transmitter){
296  while(!I2C_CheckEvent(I2Cx, I2C_EVENT_MASTER_TRANSMITTER_MODE_SELECTED));
297  }
298  else if(direction == I2C_Direction_Receiver){
299  while(!I2C_CheckEvent(I2Cx, I2C_EVENT_MASTER_RECEIVER_MODE_SELECTED));
300  }
301 }
